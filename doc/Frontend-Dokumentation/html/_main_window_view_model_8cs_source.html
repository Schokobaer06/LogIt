115 public async Task RefreshAsync()
116 {
117 // 1) Tabelle aktualisieren
118 var all = await _apiService.GetAllLogEntriesAsync();
119 var displayList = all
120 .Select(le => new LogEntryDisplay(le))
121 .OrderByDescending(d => d.IsActive)
122 .ThenByDescending(d => d.SortKey)
123 .ToList();
124
125 Entries.Clear();
126 foreach (var d in displayList)
127 Entries.Add(d);
128
129 // 2) Sessions pro Tag und Programm aufteilen
130 var sessions = new List<(string ProgramName, DateTime Day, TimeSpan Duration)>();
131 foreach (var le in all)
132 {
133 foreach (var s in le.Sessions)
134 {
135 var start = s.StartTime;
136 var end = s.EndTime ?? DateTime.Now;
137 var dayStart = start.Date;
138 var dayEnd = end.Date;
139
140 for (var day = dayStart; day <= dayEnd; day = day.AddDays(1))
141 {
142 // Anfang und Ende für diesen Tag
143 var rangeStart = day == dayStart ? start : day;
144 var rangeEnd = day == dayEnd ? end : day.AddDays(1);
145
146 var duration = rangeEnd - rangeStart;
147 if (duration.TotalSeconds > 0)
148 {
149 sessions.Add((le.ProgramName, day, duration));
150 }
151 }
152 }
153 }
154
155 // Alle Tage, an denen geloggt wurde (sortiert)
156 var allDays = sessions
157 .Select(x => x.Day)
158 .Distinct()
159 .OrderBy(d => d)
160 .ToList();
161
162 // Alle Programme, die jemals geloggt wurden
163 var allPrograms = sessions
164 .Select(x => x.ProgramName)
165 .Distinct()
166 .OrderBy(p => p)
167 .ToList();
168
169 // 3) X-Achsen-Beschriftung: alle 7 Tage ein Label, sonst leer
170 Labels = allDays.Select((d, i) =>
171 i % 7 == 0 ? d.ToString("dd. MMM") : "").ToArray();
172 RaisePropertyChanged(nameof(Labels));
173
174 // 4) Matrix: Für jedes Programm, für jeden Tag die Nutzungszeit (in Stunden)
175 var values = new double[allPrograms.Count][];
176 for (int i = 0; i < allPrograms.Count; i++)
177 {
178 var prog = allPrograms[i];
179 values[i] = allDays
180 .Select(day => sessions
181 .Where(x => x.ProgramName == prog && x.Day == day)
182 .Sum(x => x.Duration.TotalHours))
183 .ToArray();
184 }
185
186 // 5) Farben für die Programme
187 var colorPalette = new[]
188 {
189 SkiaSharp.SKColors.SteelBlue, SkiaSharp.SKColors.Orange, SkiaSharp.SKColors.MediumSeaGreen,
190 SkiaSharp.SKColors.MediumVioletRed, SkiaSharp.SKColors.Goldenrod, SkiaSharp.SKColors.MediumSlateBlue,
191 SkiaSharp.SKColors.Crimson, SkiaSharp.SKColors.Teal, SkiaSharp.SKColors.DarkCyan,
192 SkiaSharp.SKColors.DarkOrange, SkiaSharp.SKColors.DarkMagenta, SkiaSharp.SKColors.DarkGreen
193 };
194
195 // 6) Series für den Chart
196 Series = allPrograms
197 .Select((prog, i) =>
198 (ISeries)new StackedColumnSeries<double>
199 {
200 Name = prog,
201 Values = values[i],
202 XToolTipLabelFormatter = point =>
203 {
204 int idx = point.Index;
205 if (idx >= 0 && idx < allDays.Count)
206 return allDays[idx].ToString("dd. MMM yyyy");
207 return "";
208 },
209 YToolTipLabelFormatter = point =>
210 {
211 var hours = point.Coordinate.PrimaryValue;
212 var ts = TimeSpan.FromHours(hours);
213 if (ts.TotalSeconds < 1) return $"{prog}: 0s";
214 if (ts.TotalHours >= 1)
215 return $"{prog}: {ts.Hours}h {ts.Minutes}min";
216 if (ts.TotalMinutes >= 1)
217 return $"{prog}: {ts.Minutes}min {ts.Seconds}s";
218 return $"{prog}: {ts.Seconds}s";
219 },
220 Fill = new LiveChartsCore.SkiaSharpView.Painting.SolidColorPaint(colorPalette[i % colorPalette.Length]),
221 Stroke = null
222 })
223 .ToArray();
224 RaisePropertyChanged(nameof(Series));
225
226 // 7) Achsen setzen
227 XAxes = new[]
228 {
229 new Axis
230 {
231 Labels = Labels,
232 LabelsRotation = 0,
233 MinStep = 1,
234 Name = "Tag",
235 TextSize = 14,
236 Padding = new LiveChartsCore.Drawing.Padding(10),
237 NamePaint = new SolidColorPaint(SKColors.White), // Überschrift
238 LabelsPaint = new SolidColorPaint(SKColors.LightGray) // Achsenbeschriftung
239 }
